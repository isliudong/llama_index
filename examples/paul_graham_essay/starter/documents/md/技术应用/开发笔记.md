# 修复个人发起方流程	
UPDATE hesg_flow hf
INNER JOIN hesg_flow_signatory hfs ON hfs.sponsor_flag = 1 
AND hfs.signatory_type = "PERSONAL" 
AND hf.flow_id = hfs.flow_id 
SET 
hf.sponsor_agent_name = hfs.signatory_name,
hf.sponsor_type='PERSONAL',
hf.sponsor_agent_phone = hfs.receiver_phone;

UPDATE hesg_flow hf
INNER JOIN hesg_person hp ON
hf.sponsor_type="PERSONAL"
AND hp.name = hf.sponsor_agent_name
AND hp.phone = hf.sponsor_agent_phone
AND hf.tenant_id = hp.tenant_id 
SET 
hf.sponsor_id = hp.person_id;


# 修复企业发起方流程	
UPDATE hesg_flow hf
INNER JOIN hesg_flow_signatory hfs ON hfs.sponsor_flag = 1

AND hf.flow_id = hfs.flow_id

hf.sponsor_company_name=hfs.signatory_name,
